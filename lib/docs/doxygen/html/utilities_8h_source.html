55  /**
56  * \brief Rotate the given bits to the left and return the result.
57  *
58  * \param bits
59  * The value to rotate.
60  * \param count
61  * The amount of bits to rotate (shift to left and add to right).
62  * \param size
63  * The size in bits used in `bits`.
64  * <BR>e.g. 1 byte given => size = 8
65  * <BR>e.g. only 6 bits in use => size = 6, and all bits higher than (1 << size-1) will be zeroed.
66  */
67  inline uint32_t rotl(uint32_t bits, uint32_t count = 1u, const uint32_t size = 8u) {
68  const uint32_t len_mask = (1u << size) - 1u;
69 
70  count %= size; // Limit bit rotate count to size
71  bits &= len_mask; // Limit given bits to size
72 
73  return ((bits << count) & len_mask) | (bits >> (size - count));
74  }

76  /**
77  * \brief Return the `v` represented in a binary string.
78  *
79  * \tparam T
80  * The type of variable to convert.
81  * \param v
82  * The value to convert.
83  * \param bitwidth
84  * The length in bits of the given variable `v`.
85  */
86  template <typename T>
87  inline std::string to_bin(const T v, const uint32_t bitwidth) {
88  #ifdef REV_BITS
89  const uint64_t maxpow = bitwidth ? (1ull << (bitwidth - 1)) : 0;
90  uint64_t mask;
91 
92  std::string result = "";
93 
94  for (mask = 0x1; mask <= maxpow; mask <<= 1) {
95  result += (v & mask) ? "1" : "0";
96  }
97  #else
98  uint64_t mask = bitwidth ? (1ull << bitwidth) : 0;
99 
100  std::string result = "";
101 
102  while(mask >>= 1) {
103  result += (v & mask) ? "1" : "0";
104  }
105  #endif
106 
107  return result;
108  }

135  * \brief Check whether the parity of the given binary string is even.
136  *
137  * \param word
138  * The string to check.
139  * \param even
140  * Check for even (`true`) or uneven (`false`) parity.

142  inline bool check_parity_string(const std::string& word, const bool even = true) {
143  size_t count = 0, i = 0;
144 
145  while(i < 7) {
146  if (word[i++] == '1')
147  ++count;
148  }
149 
150  return (count & 0x1) == !even;

153  /**
154  * \brief Check whether the parity of the given uint64_t is even.
155  * <BR>See https://graphics.stanford.edu/~seander/bithacks.html for more.
156  *
157  * \param word
158  * The uint64_t to check.
159  * \param even
160  * Check for even (`true`) or uneven (`false`) parity.
161  */
162  inline bool check_parity(uint64_t word, const bool even = true) {
163  word ^= word >> 1;
164  word ^= word >> 2;
165  word = (word & 0x1111111111111111UL) * 0x1111111111111111UL;
166 
167  return ((word >> 60ull) & 1ull) == !even;
168  }

171  * \brief Select the bits in data given by the indices in `*indices`.
172  *
173  * \param data
174  * The data to select bits from.
175  * \param *indices
176  * Array with the indices to select.
177  * \param n
178  * The amount of indices.

180  inline uint32_t select_bits(const uint32_t data, const uint8_t *indices, const uint8_t n) {
181  uint32_t r = 0u;
182 
183  for(uint8_t i = 0u; i < n; ++i)
184  r |= (data & (1u << indices[i])) ? (1u << i) : 0u;
185 
186  return r;

207  inline uint8_t pack_byte(const uint8_t a, const uint8_t b, const uint8_t c, const uint8_t d,

222  /**
223  * \brief Encode the given word with standard Hamming(7,4) and return a byte with the set parity bits.
224  *
225  * \param v
226  * The nibble to encode.
227  */
228  inline uint8_t hamming_encode_soft(const uint8_t v) {
229  const uint8_t p1 = bit(v, 1) ^ bit(v, 2) ^ bit(v, 3);
230  const uint8_t p2 = bit(v, 0) ^ bit(v, 1) ^ bit(v, 2);
231  const uint8_t p3 = bit(v, 0) ^ bit(v, 1) ^ bit(v, 3);
232  const uint8_t p4 = bit(v, 0) ^ bit(v, 2) ^ bit(v, 3);
233 
234  return pack_byte(p1, bit(v, 0), bit(v, 1), bit(v, 2), p2, bit(v, 3), p3, p4);
235  }

290  if (syndrome) {

292  }
293
